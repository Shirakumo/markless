\defineidentifier{compound-bold}{bold}
\definestyle{compound-bold}{font-weight: bold} \\

If given, this option marks the \g{style} to bold the \g{text}.\\

\begin{examples}
  \example{Not again(in bold)!}{Not \textbf{again}!}
\end{examples}
